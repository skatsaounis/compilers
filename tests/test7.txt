def bool quicksort( int[] a; int m,n ):
	int i, j, ii, jj, temp
	if n <=m: return  true end

	for i := m ; i <= j ; i := i+1:
		for ii:=i; a[ii] < a[(m+n) / 2]; ii := ii+1 :
			skip
		end
		for jj := j ; a[jj] > a[(m+n) / 2] ; jj := jj-1 :
			skip
		end
		if (ii <= jj):
			temp := a[ii]
			a[ii] := a[jj]
			a[jj] := temp
			ii := ii+1
			jj := jj-1
			i := ii
			j := jj
		end
	end
	quicksort(a,m,j)
	quicksort(a,i,n)
end
